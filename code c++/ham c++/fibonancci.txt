#include <stdio.h>
#include <math.h>
// kiem tra fibonanci
int checkfibo(long long n){
	if (n == 1 || n == 0){
		return 1;
	}
	long long fn1 = 1, fn2 = 0;
	for (int i = 2; i <= 92; i++){
		long long fn = fn1 + fn2;
		if (fn == n){
			return 1;
		}
		fn2 = fn1;
		fn1 = fn;
	}
	return 0;
}
// in ra n so fibo dau tien
void infibo(int n){
	if (n == 1){
		printf("0");
	}
	else if (n == 2){
		printf("0 1");
	}
	else{
		printf("0 1 ");
		long long fn1 = 1, fn2 = 0;
		for (int i = 2; i < n; i++){
			long long fn = fn1 + fn2;
			printf("%lld ", fn);
			fn2 = fn1;
			fn1 = fn;
		}
	}
}
long long findfibo(int n){
	if (n == 0 || n == 1){
		return n;
	}
	long long fn1 = 1, fn2 = 0, fn;
	for (int i = 2; i <= n; i++){
		fn = fn1 + fn2;
		fn2 = fn1;
		fn1 = fn;
	}
	return fn;
}
int main (){
	long long n;
	scanf("%lld\n", &n);
	printf("%d\n", checkfibo(n));
	infibo(n);
	printf("%lld\n", findfibo(n));
	return 0;
}
